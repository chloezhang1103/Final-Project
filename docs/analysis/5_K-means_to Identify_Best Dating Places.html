from sklearn.cluster import KMeans
from sklearn.preprocessing import StandardScaler
from sklearn.model_selection import train_test_split
from sklearn.linear_model import LinearRegression
from sklearn.metrics import mean_squared_error, r2_score
import matplotlib.pyplot as plt
import seaborn as sns
dating_score = pd.read_csv("data/DatingScore.csv")
features = [
    'nearest_parking_distance', 'bus_count_500m', 'subway_count_500m',
    'public_transport_score', 'nearest_romantic_distance',
    'noise_count', 'Median_Household_Income', 'price_cleaned'
]

factors_clean = factors.dropna(subset=features).copy()
scaler = StandardScaler()
X_scaled = scaler.fit_transform(factors_clean[features])

inertia = []
K_range = range(1, 10)

for k in K_range:
    kmeans = KMeans(n_clusters=k, random_state=42)
    kmeans.fit(X_scaled)
    inertia.append(kmeans.inertia_)
plt.figure(figsize=(8, 5))
plt.plot(K_range, inertia, marker='o')
plt.title("Elbow Method for Optimal K")
plt.xlabel("Number of Clusters (k)")
plt.ylabel("Inertia")
plt.grid()
plt.show()
k_optimal = 3  
kmeans = KMeans(n_clusters=k_optimal, random_state=42)
factors_clean['Cluster'] = kmeans.fit_predict(X_scaled)

factors_with_clusters = factors_clean[['title', 'Cluster', 'geometry'] + features]
merged_data = pd.merge(dating_score, factors_with_clusters, on='title', how='left')
merged_gdf = gpd.GeoDataFrame(merged_data, geometry='geometry')


cluster_summary = merged_gdf.groupby('Cluster').agg({
    'DatingScore': ['mean', 'count'],
    'nearest_parking_distance': 'mean',
    'bus_count_500m': 'mean',
    'subway_count_500m': 'mean',
    'noise_count': 'mean',
    'Median_Household_Income': 'mean',
    'price_cleaned': 'mean'
}).reset_index()

cluster_summary.columns = ['Cluster', 'Avg_Dating_Score', 'Count', 
                           'Avg_Parking_Distance', 'Avg_Bus_Count',
                           'Avg_Subway_Count', 'Avg_Noise_Count',

import folium
from folium.plugins import MarkerCluster

merged_gdf = merged_gdf.to_crs(epsg=4326)

merged_gdf['latitude'] = merged_gdf.geometry.y
merged_gdf['longitude'] = merged_gdf.geometry.x


cluster_colors = ['red', 'blue', 'green', 'purple', 'orange', 'pink', 'darkred', 'lightblue']

m = folium.Map(location=[40.7128, -74.0060], zoom_start=11, tiles="cartodb positron")

marker_cluster = MarkerCluster().add_to(m)

for _, row in merged_gdf.dropna(subset=['latitude', 'longitude']).iterrows():
    cluster = int(row['Cluster'])  # Ensure cluster is an integer
    cluster_color = cluster_colors[cluster % len(cluster_colors)]  # Assign color
    
    popup_text = f"""
    <b>Title:</b> {row['title']}<br>
    <b>Cluster:</b> {row['Cluster']}<br>
    <b>Dating Score:</b> {row['DatingScore']:.2f}<br>
    <b>Avg Noise:</b> {row['noise_count']:.2f}<br>
    <b>Income:</b> ${row['Median_Household_Income']:.2f}
    """
    folium.CircleMarker(
        location=[row['latitude'], row['longitude']],
        radius=6,
        color=cluster_color,
        fill=True,
        fill_opacity=0.8,
        popup=folium.Popup(popup_text, max_width=300)
    ).add_to(marker_cluster)

Cluster Characteristics
Cluster 1 High Avg_Dating_Score, moderate noise, high bus and subway counts, and high prices. These may represent premium dating spots with excellent access to transit.
Cluster 2 Moderate Avg_Dating_Score, high noise count, moderate parking distance, and relatively lower prices. These may represent busy and accessible areas.
Cluster 0 Lower Avg_Dating_Score, short parking distances, lower transit counts, and lower prices. These may represent budget-friendly spots in quieter areas.

Through this study, we provide actionable insights into the characteristics that define an ideal dating spot by analyzing factors such as accessibility, ambiance, and affordability. The results identify three distinct types of dating spots: premium locations with high dating scores, excellent transit access, and moderate noise; busy and accessible areas offering a balance of affordability and convenience; and budget-friendly spots in quieter neighborhoods with lower costs and limited transit options. These insights help pinpoint restaurants that align with varying preferences, enabling couples to choose spots that best suit their dating experiences while offering valuable recommendations for businesses aiming to create attractive and romantic environments.
